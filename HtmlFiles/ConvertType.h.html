	std::string Convert::convertContent(std::string fname) 
{
		std::string temp = "";
		std::string content = "";
		std::ifstream in(fname);
		if (!in.good())
			throw(std::exception(("can't open source file " + fname).c_str()));
		std::ostringstream out;
		out << in.rdbuf();
		temp = out.str();
		for (int i = 0; i < temp.size(); i++) 
{
			if (temp[i] == '<') 
{
				content = content + "<";
			}

			else if (temp[i] == '>') 
{
				content = content + ">";
			}

			else
				content = content + temp[i];
		}

		return content;
	}

}
